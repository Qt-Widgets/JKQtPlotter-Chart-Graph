81 /** \brief C++11 finally construct (executes a callable-object when the edestructor is executed)
82  * \ingroup jkqtptools_codestructuring
83  *
84  * Typical usage:
85  * \code
86  * {
87  * // the instruction 'painter.restore()' will be executed at the end
88  * // of the block, when __finalpaint is destroyed (see (*) below)
89  * JKQTPFinalAct __finalpaint([&painter]() { painter.restore(); });
90  *
91  * // ...
92  * // do something ...
93  * // ...
94  *
95  * } // (*) 'painter.restore()' is executed before the end of this block!
96  * \endcode
97  *
98  * \see JKQTPFinally()
99  */
100 template <class F>
101 class JKQTPFinalAct
102 {
103 public:
104  explicit JKQTPFinalAct(F f) noexcept
105  : f_(std::move(f)), invoke_(true) {}
106 
107  JKQTPFinalAct(JKQTPFinalAct&& other) noexcept
108  : f_(std::move(other.f_)),
109  invoke_(other.invoke_)
110  {
111  other.invoke_ = false;
112  }
113 
114  JKQTPFinalAct(const JKQTPFinalAct&) = delete;
115  JKQTPFinalAct& operator=(const JKQTPFinalAct&) = delete;
116 
117  ~JKQTPFinalAct() noexcept
118  {
119  if (invoke_) f_();
120  }
121 
122 private:
123  F f_;
124  bool invoke_;
125 };
